200 int Paging<T>::internalIterate(std::function<bool (T, int)> iterator, int to, int from) const
201 {
202  //handle all items in this paging
203 
204  auto offset = d->iPaging->offset();
205  auto count = d->data.size();
206  auto start = 0;
207  auto max = count;
208  if(offset >= 0) {// has indexes
209  //from
210  start = qMax(from, offset) - offset;
211  //to
212  if(to >= 0)
213  max = qMin(to, max + offset) - offset;
214  }
215 
216  //delete unused items caused by from
217  for(auto j = 0; j < start; j++)
218  MetaComponent<T>::deleteLater(d->data.value(j));
219 
220  //iterate over used items
221  int i;
222  auto canceled = false;
223  for(i = start; i < max; i++) {
224  auto item = d->data.value(i);
225  auto index = offset >= 0 ? i + offset : -1;
226  if(!iterator(item, index)) {
227  canceled = true;
228  break;
229  }
230  }
231 
232  //delete all unused items caused by to
233  for(auto j = i; j < count; j++)
234  MetaComponent<T>::deleteLater(d->data.value(j));
235 
236  if(canceled)
237  return -1;
238  else if(offset >= 0)
239  return offset + i;
240  else
241  return 0;
242 }
