194 void GenericRestReplyAwaitable<DataClassType, ErrorClassType>::prepare(const std::function<void ()> &resume)
195 {
196  reply->onSucceeded([this, resume](int, DataClassType data) {
197  errorResult.reset();
198  successResult = std::move(data);
199  resume();
200  });
201  reply->onFailed([this, resume](int code, ErrorClassType data) {
202  errorResult.reset(new exceptionType{code, RestReply::FailureError, std::move(data)});
203  resume();
204  });
205  reply->onSerializeException([this, resume](const QJsonSerializerException &data) {
206  errorResult.reset(new exceptionType{0, RestReply::DeserializationError, QString::fromUtf8(data.what())});
207  resume();
208  });
209  reply->onError([this, resume](QString message, int code, RestReply::ErrorType type) {
210  errorResult.reset(new exceptionType{code, type, std::move(message)});
211  resume();
212  });
213 }
