237 void RestReplyPrivate::replyFinished()
238 {
239  retryDelay = -1;
240  //check json first to allow data for certain network fails
241  auto readData = networkReply->readAll();
242  QJsonParseError jError;
243  auto jDoc = QJsonDocument::fromJson(readData, &jError);
244  if(jError.error != QJsonParseError::NoError && !readData.isEmpty())
245  emit q->error(jError.errorString(), jError.error, RestReply::JsonParseError, {});
246  else {
247  QJsonValue jValue;
248  if(jDoc.isObject())
249  jValue = jDoc.object();
250  else if(jDoc.isArray())
251  jValue = jDoc.array();
252 
253  //check "http errors", becaus they can have data
254  auto status = networkReply->attribute(QNetworkRequest::HttpStatusCodeAttribute).toInt();
255  if(status >= 300)//errors begin at 300
256  emit q->failed(status, jValue, {});
257  else {//now network errors
258  if(networkReply->error() != QNetworkReply::NoError)
259  emit q->error(networkReply->errorString(), networkReply->error(), RestReply::NetworkError, {});
260  else {//no errors, completed!
261  emit q->succeeded(status, jValue, {});
262  retryDelay = -1;
263  }
264  }
265  }
266 
267  if(retryDelay == 0) {
268  retryDelay = -1;
269  retryReply();
270  } else if(retryDelay > 0) {
271  QTimer::singleShot(retryDelay, Qt::PreciseTimer, this, &RestReplyPrivate::retryReply);
272  retryDelay = -1;
273  } else if(autoDelete)
274  q->deleteLater();
275 }

285 void RestReplyPrivate::retryReply()
286 {
287  auto nam = networkReply->manager();
288  auto request = networkReply->request();
289  auto verb = networkReply->property(PropertyVerb).toByteArray();
290  if(verb.isEmpty())
291  verb = "GET";
292  auto buffer = networkReply->property(PropertyBuffer).value<QIODevice*>();
293  if(buffer)
294  buffer = cloneDevice(buffer);
295 
296  networkReply->deleteLater();
297  networkReply = compatSend(nam, request, verb, buffer);
298  connectReply(networkReply);
